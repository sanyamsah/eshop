class Solution
{
    public String multiplyStrings(String s1,String s2)
    {
        //code here.
        int l1=s1.length();
        int l2=s2.length();
        int[] res=new int[l1+l2];
        int pf=0;
        for(int i=l2-1;i>=0;i--){
            int inum=s2.charAt(i) - '0';
            int j=l1-1;
            int carry=0;
            int k=res.length-1-pf;
            // for(int j=l1-1;j>=0;j--){

            //     int prod=num1*num2+carry+res[k];
            //     carry=prod/10;
            //     res[k]=prod%10;
            //     k--;
            // }
            while(j>=0 || carry != 0){
                int jnum = j>=0 ? s1.charAt(j) - '0' : 0;
                j--;
                int prod = inum * jnum + carry + res[k];
                res[k]=prod%10;
                carry=prod/10;
                k--;
            }
            pf++;
        }
        int flag=0,count=0;
        for(int i=0;i<res.length;i++){
            if(res[i]==0 && flag ==0){
                count++;
                continue;
            }
            else{
                flag=1;
                break;
            }
        }
        // int size=res.length-count;
        // char[] result = new char[size];
        // int k=count;
        // for(int i=0;i<result.length;i++){
        //     result[i]=(char)res[k];
        //     k++;
        // }
        // return result.length == 0 ? "0" : result.toString();
        //-------------------------------------------------------------------
        // String result="";
        // int flag=0;
        // for(int i : res){
        //     if(i==0 && flag==0){
        //         continue;
        //     }
        //     else{
        //         flag=1;
        //         result += res;
        //     }
        // }
        // return result;
        //-----------------------------------------------------------------------
        StringBuilder result = new StringBuilder();
        for (int num : res) {
            if (!(result.length() == 0 && num == 0)) {
                result.append(num);
            }
        }

        return result.length() == 0 ? "0" : result.toString();
    }
}
